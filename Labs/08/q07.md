```asm
INCLUDE Irvine32.inc

.data
	weekdays BYTE
    "Monday",0,0,0,0,
    "Tuesday",0,0,0,
    "Wednesday",0,
    "Thursday",0,0,
    "Friday",0,0,0,0,
    "Saturday",0,0,
    "Sunday",0,0,0,0
.code
main PROC
	call ReadInt
	dec eax
	mov ebx, 10
	imul ebx

	mov esi, eax
	mov ecx, 10
print_word:
	movzx eax, weekdays[esi]
	call WriteChar
	inc esi
	loop print_word

	exit
main ENDP
END main
```
