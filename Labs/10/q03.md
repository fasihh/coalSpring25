```asm
INCLUDE Irvine32.inc

.data
    str1 BYTE "Hello World!",0
    str2 BYTE "Hello Welt!",0
    msg1 BYTE "src > dst",0
    msg2 BYTE "src = dst",0
    msg3 BYTE "src < dst",0
.code

IsCompare PROC
    enter 0,0

    mov esi, [ebp + 8]
    mov edi, [ebp + 12]
    mov ecx, [ebp + 16]
    repe cmpsb

    leave
    ret
IsCompare ENDP

main PROC
    push LENGTHOF str1
    push OFFSET str2
    push OFFSET str1
    call IsCompare
    je _equal
    jl _lessr
    mov edx, OFFSET msg1
    jmp _end
_equal:
    mov edx, OFFSET msg2
    jmp _end
_lessr:
    mov edx, OFFSET msg3
_end:
    call WriteString
    exit
main ENDP
END main
```
